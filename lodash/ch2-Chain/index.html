_(value): 通过lodash的构造方法来包装对象使其拥有lodash的方法

	var wrapped = _([1, 2, 3]);

	// returns an unwrapped value
	wrapped.reduce(function(total, n) {
	  return total + n;
	});
	// → 6

	// returns a wrapped value
	var squares = wrapped.map(function(n) {
	  return n * n;
	});

	_.isArray(squares);
	// → false

	_.isArray(squares.value());
	// → true

_.chain(value): 创建lodash链对象

	var users = [
	  { 'user': 'barney',  'age': 36 },
	  { 'user': 'fred',    'age': 40 },
	  { 'user': 'pebbles', 'age': 1 }
	];

	var youngest = _.chain(users)
	  .sortBy('age')
	  .map(function(chr) {
		return chr.user + ' is ' + chr.age;
	  })
	  .first()
	  .value();
	// → 'pebbles is 1'

_.tap(value, interceptor, [thisArg]): 在链操作中加入拦截器
